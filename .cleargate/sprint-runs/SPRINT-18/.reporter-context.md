// Header: purpose, args, exit codes, fallback chain (see Resilience block below).
import { parseJsonl, normalizeRow, aggregate } from './lib/ledger-digest.mjs';

function main() {
  const [sprintId, ...flags] = process.argv.slice(2);
  if (!sprintId) { /* usage; exit 2 */ }
  const json = flags.includes('--json');
  const sprintDir = process.env.CLEARGATE_SPRINT_DIR
    ? path.resolve(process.env.CLEARGATE_SPRINT_DIR)
    : path.join(REPO_ROOT, '.cleargate', 'sprint-runs', sprintId);
  const ledgerFile = path.join(sprintDir, 'token-ledger.jsonl');

    process.stderr.write(`Error: token-ledger.jsonl not found at ${ledgerFile}\n`);
    process.exit(1);   // Hard error: missing file (per story R4).
  }
  const rows = parseJsonl(ledgerFile).map(normalizeRow);
  if (rows.length === 0) {
    // Empty-ledger soft path (per story §1.4 Open Q #1 — accept recommended).
    if (json) {

    process.stdout.write(JSON.stringify(digest));
  } else {
    // Human format matching CR-021 §3.2.5 example.

  process.exit(0);
}

#### Gotchas surfaced from code inspection

1. **`token-ledger.jsonl` rows have BOTH `story_id` and `work_item_id`.** Verified `.cleargate/sprint-runs/SPRINT-17/token-ledger.jsonl:1` — `"story_id":"","work_item_id":"BUG-004"`. Story R-Risk #2 says "schema-tolerant parser — accept `work_item_id || work_item || null`". The empty-string `story_id` case is real and must collapse to null in the aggregator (otherwise the `by_work_item` map gets a `""` key bucket). Fix in `normalizeRow`: `row.work_item_id || row.work_item || row.story_id || null` AND treat empty string as null: use `(row.work_item_id || row.work_item || row.story_id || '').trim() || null`.

2. **Token ledger SubagentStop attribution Red is in-flight.** Sprint Plan §3 risk row 1: rows in SPRINT-17/SPRINT-18 may be attributed to the orchestrator session, not the subagent. Story §1.3 explicitly puts the Red OUT of scope: "this story only **reads** what the ledger contains; it does not fix the upstream attribution." The aggregator MUST NOT crash when `agent_type` is missing or set to a value outside `{architect, developer, qa, reporter}` — bucket unknown types into `by_agent.unknown` (already covered by `normalizeRow` fallback to `'unknown'`).

3. **`git log --stat sprint/S-NN ^main`** silently fails when the branch is absent (story §3.2 step 3 says "exec ... 2>/dev/null" + "silent fail if branch absent"). Use `child_process.execSync` with `{ stdio: ['ignore', 'pipe', 'ignore'] }` AND wrap in try/catch — `execSync` throws on non-zero exit. 4. **Sprint-plan filename is NOT deterministic.** The plan file at `.cleargate/delivery/pending-sync/SPRINT-18_Prepare_Close_Observe_Mechanics.md` has a slug after the sprint-id prefix. Search by glob: `pending-sync/<sprint-id>_*.md` THEN fall back to `archive/<sprint-id>_*.md`. If both match (unlikely — pending-sync is the source of truth pre-close), prefer pending-sync.

// New helper (insert between atomicWriteString and invokeScript at ~line 95)
/**
 * Compute the v2-style report filename for a sprint dir.
 * Strips "SPRINT-" prefix from the basename; preserves digit-count exactly
 * (no padding/normalization — SPRINT-018 → "SPRINT-018_REPORT.md").
 * Backwards-compat: if basename does not match ^SPRINT-\d+, returns "REPORT.md"
 * unchanged. This carve-out keeps Scenario-5 ("S-99") and any legacy sprint
 * dirs operating under the old naming; the v2.1 naming-convention check at
 * line ~196 is the gate that surfaces non-conformant sprint dirs separately.
 * @param {string} sprintDirPath - absolute path to the sprint dir
 * @returns {string} - report filename relative to sprintDirPath
 */
function reportFilename(sprintDirPath) {
  const name = path.basename(sprintDirPath);
  const m = name.match(/^SPRINT-(\d+)$/);
  return m ? `SPRINT-${m[1]}_REPORT.md` : 'REPORT.md';
}
